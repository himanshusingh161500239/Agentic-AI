# Python code to calculate the approximation of Pi using a series

def approximate_pi(num_terms):
    """
    Calculates an approximation of Pi using the series:
    4 * (1 - 1/3 + 1/5 - 1/7 + ...)

    Args:
        num_terms: The number of terms to include in the series.

    Returns:
        An approximation of Pi.
    """

    pi_approx = 0.0
    for i in range(num_terms):
        term = 1.0 / (2 * i + 1)
        if i % 2 == 0:
            pi_approx += term
        else:
            pi_approx -= term

    return 4 * pi_approx
